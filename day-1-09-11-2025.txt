CI CD

CI: Contiuous Integration
CD: Contimous Deployment

pipeline:

	In Software, a series of automated steps that runs in a seq to build, test and deploy our code.

Ex: Car Assembly
	Build stage
	Test Stage


code::

stages:
  - build
  - test


build_car:
  image: alpine
  stage: build
  script:
    - echo "Building the car"
    - mkdir build
    - cd build 
    - touch car.txt
    - echo "Chassis" > car.txt
    - cat car.txt
  artifacts:
    paths:
      - build/

test_car:
  image: alpine
  stage: test
  script:
    - test -f build/car.txt   
     # using test -f to find whether we have the file in the desired location or not. 
    - grep "Chassis" build/car.txt
    # grep -> powerful text- search tool used to search for patterns within text
